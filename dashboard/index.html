🔧 SDK & API Reference

      

        
API Endpoint

        

          POST https://api.agentguard.tech/api/v1/evaluate
          
Send any agent action to this endpoint to get a real-time policy decision.

        

      

      

        
Integration Pattern
Use a runtime agent key in code

        

          

            
            
          

          

        

      

      

        

          
cURL

          

            
export AGENTGUARD_API_KEY="ag_agent_..."

curl -X POST https://api.agentguard.tech/api/v1/evaluate \
  -H "X-API-Key: $AGENTGUARD_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "tool": "http_post",
    "params": {
      "destination": "https://evil.io"
    }
  }'

          

        

        

          
Python

          

            
import os
import requests

r = requests.post(
    "https://api.agentguard.tech/api/v1/evaluate",
    headers={"X-API-Key": os.environ["AGENTGUARD_API_KEY"]},
    json={
        "tool": "http_post",
        "params": {"destination": url}
    }
)
decision = r.json()
if decision["result"] == "block":
    raise PolicyViolation(decision["reason"])

          

        

        

          
TypeScript

          

            
const res = await fetch(
  "https://api.agentguard.tech/api/v1/evaluate",
  {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "X-API-Key": process.env.AGENTGUARD_API_KEY!
    },
    body: JSON.stringify({
      tool: "http_post",
      params: { destination: url }
    })
  }
);
const { result, riskScore } = await res.json();

          

        

        

          
Response Format

          

            
{
  "result": "block",          // allow | block | monitor | require_approval
  "matchedRuleId": "block-external-http",
  "riskScore": 75,            // 0-1000
  "reason": "Blocked by rule...",
  "durationMs": 0.49          // sub-millisecond
}
