rules:
  # Block all data exfiltration
  - id: block-data-exfiltration
    action: block
    conditions:
      tool: [http_post, http_put, http_patch]
      destination: "!*.company.internal"
      severity: critical

  # Require human approval for financial ops
  - id: require-approval-financial
    action: require_approval
    conditions:
      tool: [transfer_funds, create_invoice]
      amount: "> $10,000"
      timeout_seconds: 300

  # Allow internal reads unconditionally
  - id: allow-internal-reads
    action: allow
    conditions:
      tool: [file_read, database_select]
      scope: internal
